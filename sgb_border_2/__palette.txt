sgb_palette:
	call sgb_gbc_mode			; sgb only
	jr c, @@exit


; ------------------------------------------------------------
; ------------------------------------------------------------


	call @palette_colors


	ld hl, @pal_trn
	call sgb_upload


; -------------------------------------------------------
; -------------------------------------------------------


@@exit:
	ld a, (0xd6a6)
	jp 0x385+1


; =======================================================
; =======================================================


@pal_trn:
	db 0x59, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00



; ////////////////////////////////////////////////////////
; ////////////////////////////////////////////////////////



@palette_colors:
	ld a, int(orga(sgb_palette_colors) / 0x4000)

	ld hl, sgb_palette_colors
	push hl

	jp 0x385+1
