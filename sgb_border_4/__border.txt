sgb_border:
	call @border_tiles1

	ld hl, @sgb_chr_trn1
	call sgb_upload


; -------------------------------------------------------
; -------------------------------------------------------


	call @border_tiles2

	ld hl, @sgb_chr_trn2
	call sgb_upload


; -------------------------------------------------------
; -------------------------------------------------------


	call @border_tilemap

	ld hl, @sgb_pct_trn
	call sgb_upload


; -------------------------------------------------------
; -------------------------------------------------------


	ld a, (0xcca3)
	jp 0x49f+1


; ========================================================
; ========================================================


@sgb_chr_trn1:
	.db 0x99, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00


@sgb_chr_trn2:
	.db 0x99, 0x01, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00


@sgb_pct_trn:
	.db 0xa1, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00



; ////////////////////////////////////////////////////////
; ////////////////////////////////////////////////////////



@border_tiles1:
	ld a, int(orga(sgb_border_tiles1) / 0x4000)

	ld hl, sgb_border_tiles1
	push hl

	jp 0x49f+1


; -------------------------------------------------------
; -------------------------------------------------------


@border_tiles2:
	ld a, int(orga(sgb_border_tiles2) / 0x4000)

	ld hl, sgb_border_tiles2
	push hl

	jp 0x49f+1


; -------------------------------------------------------
; -------------------------------------------------------


@border_tilemap:
	ld a, int(orga(sgb_border_tilemap) / 0x4000)

	ld hl, sgb_border_tilemap
	push hl

	jp 0x49f+1
