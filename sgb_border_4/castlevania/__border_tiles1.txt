sgb_border_tiles1:
	ld de, @data					; rom src
	ld hl, 0x8800					; vram dst
	ld bc, 0x1000



@@loop:
	ld a, (de)
	ldi (hl), a

	inc de
	dec bc

	ld a, b
	or a, c
	jr nz, @@loop



	ld a, (0xcca3)
	jp 0x49f+1



; /////////////////////////////////////////////////////////////////
; /////////////////////////////////////////////////////////////////
