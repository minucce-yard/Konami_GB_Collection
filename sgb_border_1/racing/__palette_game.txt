sgb_palette_game:
	ldh a, (0x8e)				; sgb only
	cp a, 0x11
	jr z, @@exit


; -------------------------------------------------------
; -------------------------------------------------------


	ld hl, @pal_game			; pal_set  [0a]
	call sgb_command_full


; -------------------------------------------------------
; -------------------------------------------------------


@@exit:
	call 0x54d					; old code
	ld bc, 0x2000
	call 0xead



	ld a, int(0x187c7 / 0x4000)

	ld hl, 0x47c7
	push hl

	jp 0xb27+1


; ==================================================================
; ==================================================================


@pal_game:
