sgb_palette:
	call sgb_gbc_mode			; sgb only
	jr c, @@exit


; ------------------------------------------------------------
; ------------------------------------------------------------


	call @palette_colors


	ld hl, @pal_trn
	call sgb_upload


; ------------------------------------------------------------
; ------------------------------------------------------------


	ld hl, @pal_set				; konami logo
	call sgb_command_full


; -------------------------------------------------------
; -------------------------------------------------------


@@exit:
	ldh a, (0xfe)
	jp 0xb27+1


; =======================================================
; =======================================================


@pal_trn:
	db 0x59, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00


@pal_set:
	db 0x51, 0x0c, 0x00, 0x0d, 0x00, 0x0e, 0x00, 0x0f, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00



; ////////////////////////////////////////////////////////
; ////////////////////////////////////////////////////////



@palette_colors:
	ld a, int(orga(sgb_palette_colors) / 0x4000)

	ld hl, sgb_palette_colors
	push hl

	jp 0xb27+1
